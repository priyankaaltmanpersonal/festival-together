- [ ] **Step 1: Write the regression test**

The bug: `deletePersonalSet` filters `scheduleSnapshot.sets` checking `setItem.canonical_set_id` but the field from the API is `id`. Add this test to `utils.test.js` to document the correct transform behavior:

```js
describe('deletePersonalSet schedule snapshot transform', () => {
  it('removes the deleted member from the matching set using id field', () => {
    const sets = [
      {
        id: 'set-abc',
        attendees: [{ member_id: 'me' }, { member_id: 'other' }],
        attendee_count: 2,
      },
      {
        id: 'set-xyz',
        attendees: [{ member_id: 'other' }],
        attendee_count: 1,
      },
    ];

    const canonicalSetId = 'set-abc';
    const myId = 'me';

    // This is the exact transform used in deletePersonalSet
    const updated = sets.map((setItem) => {
      if (setItem.id !== canonicalSetId) return setItem;
      const newAttendees = setItem.attendees.filter((a) => a.member_id !== myId);
      return { ...setItem, attendees: newAttendees, attendee_count: newAttendees.length };
    });

    expect(updated[0].attendees).toHaveLength(1);
    expect(updated[0].attendees[0].member_id).toBe('other');
    expect(updated[0].attendee_count).toBe(1);
    expect(updated[1].attendees).toHaveLength(1); // unrelated set unchanged
  });
});
```

In `App.js`, find `loadIndividual` (around line 1162). Replace the entire function with:

```js
const loadIndividual = () => {
  setActiveView('individual');
  setMoreSheetOpen(false);
  run('load individual schedules', async () => {
    if (!memberSession || !groupId) throw new Error('Need group and member session first');
    if (!isOnline) {
      if (individualSnapshot) {
        appendLog('OFFLINE: using cached individual schedules');
        return;
      }
      throw new Error('No cached individual schedules available offline');
    }
    const payload = await apiRequest({
      baseUrl: apiUrl,
      path: `/v1/groups/${groupId}/individual-schedules`,
      method: 'GET',
      sessionToken: memberSession,
    });
    setIndividualSnapshot(payload);
    setLastSyncAt(new Date().toISOString());
  });
};
```
